    static getStarlanes(allSectors) {
        // [é˛ç›ä¸ťć´žé‡Ťćž„] é™ĺ¶ćŻŹä¸ŞćźĺŚşçš„ćś€ĺ¤§ćźé—¨čżžçşżć•°é‡Ź

        // č®°ĺ˝•ćŻŹä¸ŞćźĺŚşĺ˝“ĺ‰Ťçš„čżžçşżć•°
        const degreeMap = new Map();
        allSectors.forEach(s => degreeMap.set(s.name, 0));

        const result = [];
        
        this.STATIC_LANES.forEach(pair => {
            const s1 = allSectors.find(s => s.name === pair[0]);
            const s2 = allSectors.find(s => s.name === pair[1]);
            
            if (s1 && s2) {
                const deg1 = degreeMap.get(s1.name);
                const deg2 = degreeMap.get(s2.name);
                
                // ĺŹŞćś‰ĺ˝“ä¸¤ç«Żçš„ćźé—¨ć•°é‡Źé˝ć˛ˇćś‰čľľĺ°ä¸Šé™ć—¶ďĽŚć‰Ťĺ…č®¸ĺ»şç«‹čżžćŽĄ
                if (deg1 < MAX_DEGREE && deg2 < MAX_DEGREE) {
                    result.push({
                        s1: s1,
                        s2: s2,
                        dist: Math.hypot(s1.x - s2.x, s1.y - s2.y)
                    });
                    degreeMap.set(s1.name, deg1 + 1);
                    degreeMap.set(s2.name, deg2 + 1);
                } else {
                    // console.warn(`[ćźĺ›ľç”źć] ć‹¦ćŞäş†čż‡č˝˝čżžçşż: ${s1.name}(${deg1}) <-> ${s2.name}(${deg2})`);

        // --- ĺ­¤ĺ˛›äżťćŠ¤é€»čľ‘ ---
        // çˇ®äżťč˘« MAX_DEGREE ćŞć–­çš„čŠ‚ç‚ąä¸Ťč‡łäşŽĺŹćć­»čˇĺŚ
        allSectors.forEach(s => {

                let bestTarget = null;
                let minDist = Infinity;
                
                allSectors.forEach(other => {

                    result.push({
                        s1: s,
                        s2: bestTarget,
                        dist: minDist
                    });
                    degreeMap.set(s.name, degreeMap.get(s.name) + 1);
                    degreeMap.set(bestTarget.name, degreeMap.get(bestTarget.name) + 1);
                    // console.log(`[ćźĺ›ľç”źć] ä¸şĺ­¤ĺ˛› ${s.name} ĺ»şç«‹äżťĺş•čżžçşżč‡ł ${bestTarget.name}`);
                }
            }
        });

        return result;
    }

    static getStarlanePath(startNode, endNode, allSectors) {
        const lanes = this.getStarlanes(allSectors);
        const adj = new Map();
        allSectors.forEach(s => adj.set(s.name, []));
        
        lanes.forEach(lane => {
            adj.get(lane.s1.name).push(lane.s2);
            adj.get(lane.s2.name).push(lane.s1);
        });

        const queue = [[startNode]];
        const visited = new Set([startNode.name]);

        while(queue.length > 0) {
            const path = queue.shift();
            const current = path[path.length - 1];

            if(current.name === endNode.name) {
                return path;
            }

            const neighbors = adj.get(current.name) || [];
            for(const neighbor of neighbors) {
                if(!visited.has(neighbor.name)) {
                    visited.add(neighbor.name);
                    queue.push([...path, neighbor]);
                }
            }
        }
        return null;
    }
